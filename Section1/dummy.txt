Bash Script: Port Scanner
#!/bin/bash
if [ -z "$1" ]; then
 echo "Usage: $0 <IP>"
 exit 1
fi
IP=$1
ports=(21 22 80 443 3306)
for port in "${ports[@]}"; do
 nc -zv $IP $port
done
